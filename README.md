# Angel School Manager

Role-based school management frontend connected to Supabase.

## Demo Auth IDs

Use one of these account IDs on the login screen:

- `Admin@123@gmail.com` -> role: `admin`
- `teacher@123` -> role: `teacher`
- `student@123` -> role: `student`

## Role Authority

- `admin`
  - Access: all pages
  - Create/Edit: all modules
  - Settings save: allowed
- `teacher`
  - Access: Dashboard, Students, Classes, Attendance, Homework, Announcements, Settings
  - Create: Attendance, Homework, Announcements
  - Students/Classes: read-only
  - Settings save: blocked
- `student`
  - Access: Dashboard, Homework, Announcements, Settings
  - Create: none (read-only)
  - Settings save: blocked

## Dark Mode

- Available in `Settings`.
- Persisted in browser local storage key: `angel-manager-theme`.

## Supabase Setup

1. Copy `.env.example` to `.env.local`
2. Fill:
   - `REACT_APP_SUPABASE_URL`
   - `REACT_APP_SUPABASE_PUBLISHABLE_KEY`
3. Run `npm start`

## Netlify Deploy

Create React App reads env vars at build time only.

1. Netlify -> `Site configuration` -> `Environment variables`
2. Add:
   - `REACT_APP_SUPABASE_URL`
   - `REACT_APP_SUPABASE_PUBLISHABLE_KEY`
   - `REACT_APP_SUPABASE_PRODUCTS_TABLE=products`
3. `Deploys` -> `Clear cache and deploy site`

## Full SQL (Run Once in Supabase)

```sql
create table if not exists public.students (
  id bigint generated by default as identity primary key,
  full_name text not null,
  email text,
  grade text not null,
  guardian_name text,
  status text not null default 'active',
  created_at timestamptz not null default now()
);

create table if not exists public.teachers (
  id bigint generated by default as identity primary key,
  full_name text not null,
  email text,
  subject text not null,
  phone text,
  status text not null default 'active',
  created_at timestamptz not null default now()
);

create table if not exists public.classes (
  id bigint generated by default as identity primary key,
  class_name text not null,
  section text not null,
  room_no text,
  class_teacher text,
  schedule text,
  created_at timestamptz not null default now()
);

create table if not exists public.attendance_records (
  id bigint generated by default as identity primary key,
  student_name text not null,
  class_name text not null,
  attendance_date date not null,
  status text not null default 'present',
  remarks text,
  created_at timestamptz not null default now()
);

create table if not exists public.homework (
  id bigint generated by default as identity primary key,
  title text not null,
  subject text not null,
  class_name text not null,
  due_date date not null,
  instructions text,
  status text not null default 'published',
  created_at timestamptz not null default now()
);

create table if not exists public.announcements (
  id bigint generated by default as identity primary key,
  title text not null,
  audience text not null default 'all',
  publish_date date not null default now()::date,
  message text not null,
  status text not null default 'published',
  created_at timestamptz not null default now()
);

create table if not exists public.app_settings (
  id bigint generated by default as identity primary key,
  key text not null unique,
  value_text text not null,
  updated_at timestamptz not null default now()
);

create table if not exists public.products (
  id bigint generated by default as identity primary key,
  title text not null,
  price numeric(10,2) not null default 0,
  currency text not null default 'USD',
  status text not null default 'draft',
  created_at timestamptz not null default now()
);

alter table public.students enable row level security;
alter table public.teachers enable row level security;
alter table public.classes enable row level security;
alter table public.attendance_records enable row level security;
alter table public.homework enable row level security;
alter table public.announcements enable row level security;
alter table public.app_settings enable row level security;
alter table public.products enable row level security;

drop policy if exists "public students access" on public.students;
create policy "public students access" on public.students for all using (true) with check (true);

drop policy if exists "public teachers access" on public.teachers;
create policy "public teachers access" on public.teachers for all using (true) with check (true);

drop policy if exists "public classes access" on public.classes;
create policy "public classes access" on public.classes for all using (true) with check (true);

drop policy if exists "public attendance access" on public.attendance_records;
create policy "public attendance access" on public.attendance_records for all using (true) with check (true);

drop policy if exists "public homework access" on public.homework;
create policy "public homework access" on public.homework for all using (true) with check (true);

drop policy if exists "public announcements access" on public.announcements;
create policy "public announcements access" on public.announcements for all using (true) with check (true);

drop policy if exists "public settings access" on public.app_settings;
create policy "public settings access" on public.app_settings for all using (true) with check (true);

drop policy if exists "public products access" on public.products;
create policy "public products access" on public.products for all using (true) with check (true);
```

Important: The above policies are open for speed/demo.  
For real production, replace with authenticated row-level policies.
